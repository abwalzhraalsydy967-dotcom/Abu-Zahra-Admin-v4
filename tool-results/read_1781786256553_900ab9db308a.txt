   206тЖТ   206тЖТ   206тЖТ        if (resultCode == 0 || data == null) {
   207тЖТ   207тЖТ   207тЖТ            Log.e(TAG, "Invalid permission data")
   208тЖТ   208тЖТ   208тЖТ            return false
   209тЖТ   209тЖТ   209тЖТ        }
   210тЖТ   210тЖТ   210тЖТ

   214тЖТ   214тЖТ   214тЖТ

   216тЖТ   216тЖТ   216тЖТ            mediaProjection?.registerCallback(object : MediaProjection.Callback() {
   217тЖТ   217тЖТ   217тЖТ                override fun onStop() {
   218тЖТ   218тЖТ   218тЖТ                    Log.i(TAG, "MediaProjection stopped")
   219тЖТ   219тЖТ   219тЖТ                    stopStreaming()

   227тЖТ   227тЖТ   227тЖТ                val bounds = windowManager.currentWindowMetrics.bounds
   228тЖТ   228тЖТ   228тЖТ                screenWidth = bounds.width()
   229тЖТ   229тЖТ   229тЖТ                screenHeight = bounds.height()

   231тЖТ   231тЖТ   231тЖТ            } else {
   232тЖТ   232тЖТ   232тЖТ                @Suppress("DEPRECATION")
   233тЖТ   233тЖТ   233тЖТ                windowManager.defaultDisplay.getRealMetrics(metrics)
   234тЖТ   234тЖТ   234тЖТ                screenWidth = metrics.widthPixels
   235тЖТ   235тЖТ   235тЖТ                screenHeight = metrics.heightPixels
   236тЖТ   236тЖТ   236тЖТ                screenDensity = metrics.densityDpi
   237тЖТ   237тЖТ   237тЖТ            }
   238тЖТ   238тЖТ   238тЖТ            
   239тЖТ   239тЖТ   239тЖТ            // Adjust resolution based on config
   240тЖТ   240тЖТ   240тЖТ            if (config.width < screenWidth || config.height < screenHeight) {
   241тЖТ   241тЖТ   241тЖТ                val scale = minOf(
   242тЖТ   242тЖТ   242тЖТ                    config.width.toFloat() / screenWidth,
   243тЖТ   243тЖТ   243тЖТ                    config.height.toFloat() / screenHeight
   244тЖТ   244тЖТ   244тЖТ                )
   245тЖТ   245тЖТ   245тЖТ                screenWidth = (screenWidth * scale).toInt()
   246тЖТ   246тЖТ   246тЖТ                screenHeight = (screenHeight * scale).toInt()

   248тЖТ   248тЖТ   248тЖТ            
   249тЖТ   249тЖТ   249тЖТ            Log.i(TAG, "MediaProjection initialized: ${screenWidth}x${screenHeight}")
   250тЖТ   250тЖТ   250тЖТ            return true
   251тЖТ   251тЖТ   251тЖТ            
   252тЖТ   252тЖТ   252тЖТ        } catch (e: Exception) {
   253тЖТ   253тЖТ   253тЖТ            Log.e(TAG, "Failed to initialize MediaProjection", e)
   254тЖТ   254тЖТ   254тЖТ            return false
   255тЖТ   255тЖТ   255тЖТ        }

   262тЖТ   262тЖТ   262тЖТ        if (isStreaming.get()) {
   263тЖТ   263тЖТ   263тЖТ            Log.w(TAG, "Already streaming")
   264тЖТ   264тЖТ   264тЖТ            return
   265тЖТ   265тЖТ   265тЖТ        }
   266тЖТ   266тЖТ   266тЖТ        
   267тЖТ   267тЖТ   267тЖТ        // Initialize MediaProjection
   268тЖТ   268тЖТ   268тЖТ        if (!initMediaProjection(resultCode, data)) {
   269тЖТ   269тЖТ   269тЖТ            updateStreamState(error = "Failed to initialize MediaProjection")
   270тЖТ   270тЖТ   270тЖТ            return
   271тЖТ   271тЖТ   271тЖТ        }
   272тЖТ   272тЖТ   272тЖТ        
   273тЖТ   273тЖТ   273тЖТ        // Update config with actual resolution
   274тЖТ   274тЖТ   274тЖТ        config = config.copy(
   275тЖТ   275тЖТ   275тЖТ            width = screenWidth,
   276тЖТ   276тЖТ   276тЖТ            height = screenHeight
   277тЖТ   277тЖТ   277тЖТ        )
   278тЖТ   278тЖТ   278тЖТ        
   279тЖТ   279тЖТ   279тЖТ        // Initialize encoders
   280тЖТ   280тЖТ   280тЖТ        if (!initEncoders()) {
   281тЖТ   281тЖТ   281тЖТ            updateStreamState(error = "Failed to initialize encoders")
   282тЖТ   282тЖТ   282тЖТ            cleanup()
   283тЖТ   283тЖТ   283тЖТ            return
   284тЖТ   284тЖТ   284тЖТ        }
   285тЖТ   285тЖТ   285тЖТ        
   286тЖТ   286тЖТ   286тЖТ        // Create VirtualDisplay BEFORE connecting to server
   287тЖТ   287тЖТ   287тЖТ        if (!createVirtualDisplay()) {
   288тЖТ   288тЖТ   288тЖТ            updateStreamState(error = "Failed to create VirtualDisplay")

   405тЖТ   405тЖТ   405тЖТ    /**
   406тЖТ   406тЖТ   406тЖТ     * Connect to streaming server via WebSocket.
   407тЖТ   407тЖТ   407тЖТ     * Non-blocking - always returns true. Retries automatically on failure.
   408тЖТ   408тЖТ   408тЖТ     */
   409тЖТ   409тЖТ   409тЖТ    private fun connectToServer() {
   410тЖТ   410тЖТ   410тЖТ        if (config.serverUrl.isBlank()) {
   411тЖТ   411тЖТ   411тЖТ            config = config.copy(serverUrl = Config.getBaseUrl())
   412тЖТ   412тЖТ   412тЖТ        }
   413тЖТ   413тЖТ   413тЖТ        val serverUrl = config.serverUrl.ifEmpty {
   414тЖТ   414тЖТ   414тЖТ            StreamConfig.getWebSocketUrl(this)
   415тЖТ   415тЖТ   415тЖТ        }
   416тЖТ   416тЖТ   416тЖТ        
   417тЖТ   417тЖТ   417тЖТ        try {
   418тЖТ   418тЖТ   418тЖТ            val request = Request.Builder()
   419тЖТ   419тЖТ   419тЖТ                .url(serverUrl)
   420тЖТ   420тЖТ   420тЖТ                .build()
   421тЖТ   421тЖТ   421тЖТ            
   422тЖТ   422тЖТ   422тЖТ            webSocket?.close(1000, "Reconnecting")
   423тЖТ   423тЖТ   423тЖТ            webSocket = null
   424тЖТ   424тЖТ   424тЖТ            
   425тЖТ   425тЖТ   425тЖТ            webSocket = okHttpClient?.newWebSocket(request, object : WebSocketListener() {
   426тЖТ   426тЖТ   426тЖТ                override fun onOpen(webSocket: WebSocket, response: Response) {
   427тЖТ   427тЖТ   427тЖТ                    Log.i(TAG, "WebSocket connected")
   428тЖТ   428тЖТ   428тЖТ                    wsReconnectAttempts = 0
   429тЖТ   429тЖТ   429тЖТ                    
   430тЖТ   430тЖТ   430тЖТ                    // Send stream configuration
   431тЖТ   431тЖТ   431тЖТ                    val configJson = com.google.gson.Gson().toJson(config.toMap())
   432тЖТ   432тЖТ   432тЖТ                    webSocket.send(configJson)
   433тЖТ   433тЖТ   433тЖТ                }
   434тЖТ   434тЖТ   434тЖТ                
   435тЖТ   435тЖТ   435тЖТ                override fun onMessage(webSocket: WebSocket, text: String) {
   436тЖТ   436тЖТ   436тЖТ                    handleServerMessage(text)
   437тЖТ   437тЖТ   437тЖТ                }
   438тЖТ   438тЖТ   438тЖТ                
   439тЖТ   439тЖТ   439тЖТ                override fun onClosing(webSocket: WebSocket, code: Int, reason: String) {
   440тЖТ   440тЖТ   440тЖТ                    Log.i(TAG, "WebSocket closing: $code - $reason")

   442тЖТ   442тЖТ   442тЖТ                }
   443тЖТ   443тЖТ   443тЖТ                
   444тЖТ   444тЖТ   444тЖТ                override fun onClosed(webSocket: WebSocket, code: Int, reason: String) {

   447тЖТ   447тЖТ   447тЖТ                
   448тЖТ   448тЖТ   448тЖТ                override fun onFailure(webSocket: WebSocket, t: Throwable, response: Response?) {
   449тЖТ   449тЖТ   449тЖТ                    Log.w(TAG, "WebSocket failure: ${t.message}")
   450тЖТ   450тЖТ   450тЖТ                    // Retry connection with exponential backoff
   451тЖТ   451тЖТ   451тЖТ                    if (isStreaming.get()) {
   452тЖТ   452тЖТ   452тЖТ                        scheduleReconnect(handler) { connectToServer() }
   453тЖТ   453тЖТ   453тЖТ                    }
   454тЖТ   454тЖТ   454тЖТ                }
   455тЖТ   455тЖТ   455тЖТ            })
   456тЖТ   456тЖТ   456тЖТ            
   457тЖТ   457тЖТ   457тЖТ        } catch (e: Exception) {
   458тЖТ   458тЖТ   458тЖТ            Log.e(TAG, "Failed to connect to server", e)
   459тЖТ   459тЖТ   459тЖТ            // Retry with exponential backoff
   460тЖТ   460тЖТ   460тЖТ            if (isStreaming.get()) {
   461тЖТ   461тЖТ   461тЖТ                scheduleReconnect(handler) { connectToServer() }
   462тЖТ   462тЖТ   462тЖТ            }
   463тЖТ   463тЖТ   463тЖТ        }
   464тЖТ   464тЖТ   464тЖТ    }
   465тЖТ   465тЖТ   465тЖТ    
   466тЖТ   466тЖТ   466тЖТ    /**
   467тЖТ   467тЖТ   467тЖТ     * Schedule a WebSocket reconnection with exponential backoff.
   468тЖТ   468тЖТ   468тЖТ     * Delay: base * 2^attempt, capped at wsMaxReconnectDelayMs, max maxWsReconnectAttempts retries.
   469тЖТ   469тЖТ   469тЖТ     */
   470тЖТ   470тЖТ   470тЖТ    private fun scheduleReconnect(handler: Handler, action: () -> Unit) {
   471тЖТ   471тЖТ   471тЖТ        if (wsReconnectAttempts >= maxWsReconnectAttempts) {
   472тЖТ   472тЖТ   472тЖТ            Log.e(TAG, "Max WebSocket reconnection attempts ($maxWsReconnectAttempts) reached, giving up")
   473тЖТ   473тЖТ   473тЖТ            return
   474тЖТ   474тЖТ   474тЖТ        }
   475тЖТ   475тЖТ   475тЖТ        val delay = (wsBaseReconnectDelayMs * (1L shl wsReconnectAttempts)).coerceAtMost(wsMaxReconnectDelayMs)
   476тЖТ   476тЖТ   476тЖТ        wsReconnectAttempts++
   477тЖТ   477тЖТ   477тЖТ        Log.i(TAG, "Scheduling WebSocket reconnect in ${delay}ms (attempt $wsReconnectAttempts/$maxWsReconnectAttempts)")
   478тЖТ   478тЖТ   478тЖТ        handler.postDelayed({
   479тЖТ   479тЖТ   479тЖТ            if (isStreaming.get()) action()
   480тЖТ   480тЖТ   480тЖТ        }, delay)
   481тЖТ   481тЖТ   481тЖТ    }

   484тЖТ   484тЖТ   484тЖТ     * Handle message from server
