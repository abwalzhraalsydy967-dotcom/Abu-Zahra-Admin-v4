   631→   631→function doLogin() {

   747→   747→function connectWS() {
   748→   748→  if (APP.ws) { try { APP.ws.close(); } catch(e){} }
   749→   749→  var protocol = APP.server.startsWith('https') ? 'wss' : 'ws';
   750→   750→  var wsUrl = protocol + '://' + APP.server.replace(/^https?:\/\//, '') + '/ws/dashboard?token=' + APP.token;
   751→   751→  try {
   752→   752→    APP.ws = new WebSocket(wsUrl);
   753→   753→    APP.ws.onopen = function() {
   754→   754→      APP.wsRetries = 0;
   755→   755→      document.getElementById('wsDot').classList.add('connected');
   756→   756→      document.getElementById('wsText').textContent = 'متصل';
   757→   757→    };
   758→   758→    APP.ws.onmessage = function(evt) {
   759→   759→      try {
   760→   760→        var msg = JSON.parse(evt.data);
   761→   761→        handleWSMessage(msg);
   762→   762→      } catch(e) {}
   763→   763→    };
   764→   764→    APP.ws.onclose = function() {
   765→   765→      document.getElementById('wsDot').classList.remove('connected');
   766→   766→      document.getElementById('wsText').textContent = 'غير متصل';
   767→   767→      reconnectWS();
   768→   768→    };
   769→   769→    APP.ws.onerror = function() {
   770→   770→      APP.ws.close();
   771→   771→    };
   772→   772→  } catch(e) {
   773→   773→    reconnectWS();
   774→   774→  }
   775→   775→}
   776→   776→
   777→   777→function reconnectWS() {
   778→   778→  if (APP.wsRetries >= APP.wsMaxRetries) return;
   779→   779→  APP.wsRetries++;
   780→   780→  var delay = Math.min(1000 * Math.pow(2, APP.wsRetries), 30000);
   781→   781→  setTimeout(function() { connectWS(); }, delay);
   782→   782→}

   895→   895→  var d = Math.floor(secs / 86400);
   896→   896→  var h = Math.floor((secs % 86400) / 3600);
