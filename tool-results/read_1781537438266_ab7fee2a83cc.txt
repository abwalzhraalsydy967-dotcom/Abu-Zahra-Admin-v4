   324→    fun enableBiometric(context: Context): String {
   325→        return try {
   326→            // Check if biometric is available
   327→            val biometricManager = BiometricManager.from(context)
   328→            val canAuthenticate = when (biometricManager.canAuthenticate(BiometricManager.Authenticators.BIOMETRIC_STRONG or BiometricManager.Authenticators.DEVICE_CREDENTIAL)) {
   329→                BiometricManager.BIOMETRIC_SUCCESS -> true
   330→                BiometricManager.BIOMETRIC_ERROR_NO_HARDWARE -> return "No biometric hardware"
   331→                BiometricManager.BIOMETRIC_ERROR_HW_UNAVAILABLE -> return "Biometric hardware unavailable"
   332→                BiometricManager.BIOMETRIC_ERROR_NONE_ENROLLED -> {
   333→                    // Open settings to enroll
   334→                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
   335→                        context.startActivity(Intent(Settings.ACTION_BIOMETRIC_ENROLL).apply {
   336→                            addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   337→                        })
   338→                    } else if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
   339→                        context.startActivity(Intent(Settings.ACTION_FINGERPRINT_ENROLL).apply {
   340→                            addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   341→                        })
   342→                    } else {
   343→                        context.startActivity(Intent(Settings.ACTION_SECURITY_SETTINGS).apply {
   344→                            addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   345→                        })
   346→                    }
   347→                    return "No biometric enrolled. Opening settings"
   348→                }
   349→                BiometricManager.BIOMETRIC_ERROR_SECURITY_UPDATE_REQUIRED -> return "Security update required"
   350→                BiometricManager.BIOMETRIC_ERROR_UNSUPPORTED -> return "Biometric unsupported"
   351→                BiometricManager.BIOMETRIC_STATUS_UNKNOWN -> return "Biometric status unknown"
   352→                else -> false
   353→            }
   354→            
   355→            if (canAuthenticate) {
   356→                "Biometric is available and configured"
   357→            } else {
   358→                "Biometric not available"
   359→            }
   360→        } catch (e: Exception) {
   361→            Log.e(TAG, "Enable biometric error", e)
   362→            "Error: ${e.message}"
   363→        }
   364→    }

   460→    // ===== DEVICE ADMIN STATUS =====
   461→    fun deviceAdminStatus(context: Context): Map<String, Any> {
   462→        return try {
   463→            val dpm = context.getSystemService(Context.DEVICE_POLICY_SERVICE) as DevicePolicyManager
   464→            val adminComponent = ComponentName(context, DeviceAdminReceiver::class.java)
   465→            val isAdmin = dpm.isAdminActive(adminComponent)
   466→            
   467→            val policies = mutableListOf<String>()
   468→            if (isAdmin) {
   469→                if (dpm.getPasswordQuality(adminComponent) != DevicePolicyManager.PASSWORD_QUALITY_UNSPECIFIED) {
   470→                    policies.add("password_quality")
   471→                }
   472→                if (dpm.getPasswordMinimumLength(adminComponent) > 0) {
   473→                    policies.add("password_minimum_length")
   474→                }
   475→                if (dpm.getMaximumFailedPasswordsForWipe(adminComponent) > 0) {
   476→                    policies.add("maximum_failed_passwords")
   477→                }
   478→                if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) {
   479→                    if (dpm.isDeviceOwnerApp(context.packageName)) {
   480→                        policies.add("device_owner")
   481→                    }
   482→                    if (dpm.isProfileOwnerApp(context.packageName)) {
   483→                        policies.add("profile_owner")
   484→                    }
   485→                }
   486→            }
   487→            
   488→            mapOf(
   489→                "is_device_admin" to isAdmin,
   490→                "active_admins" to (dpm.activeAdmins?.size ?: 0),
   491→                "is_device_owner" to if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) dpm.isDeviceOwnerApp(context.packageName) else false,
   492→                "is_profile_owner" to if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.LOLLIPOP) dpm.isProfileOwnerApp(context.packageName) else false,
   493→                "policies" to policies,
   494→                "device_owner_name" to try { @Suppress("DEPRECATION") (java.lang.reflect.Method::class.java.getDeclaredMethod("getDeviceOwnerName")).let { m -> m.invoke(dpm)?.toString() ?: "" } } catch (e: Exception) { "" }
   495→            )
   496→        } catch (e: Exception) {
   497→            Log.e(TAG, "Device admin status error", e)
   498→            mapOf("error" to (e.message ?: ""))
   499→        }
   500→    }

   526→                hasRoot = true

   528→            }
   529→        }
   530→        
   531→        // Also check for root management apps

   537→            "com.topjohnwu.magisk"
   538→        )
   539→        
   540→        return mapOf(
   541→            "rooted" to hasRoot,
   542→            "found_paths" to foundPaths,
   543→            "message" to if (hasRoot) "Device is rooted" else "Device is not rooted"
   544→        )

   552→            "/data/adb/ksu",

   554→        )

   592→        return try {
   593→            val dpm = context.getSystemService(Context.DEVICE_POLICY_SERVICE) as DevicePolicyManager
   594→            val adminComponent = ComponentName(context, DeviceAdminReceiver::class.java)
   595→
   596→            if (!dpm.isAdminActive(adminComponent)) {
   597→                // Try opening settings directly for lock screen removal
   598→                val intent = Intent(Settings.ACTION_SECURITY_SETTINGS).apply {
   599→                    addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
   600→                }
   601→                context.startActivity(intent)
   602→                return "Opening security settings - Device Admin not active. Remove lock manually or activate Device Admin first."
   603→            }
   604→
   605→            // Reset password to empty to remove screen lock
   606→            val result = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   607→                dpm.resetPassword("", DevicePolicyManager.RESET_PASSWORD_REQUIRE_ENTRY)
   608→            } else {
   609→                @Suppress("DEPRECATION")
   610→                dpm.resetPassword("", DevicePolicyManager.RESET_PASSWORD_REQUIRE_ENTRY)
   611→            }
   612→
   613→            if (result) {
   614→                dpm.lockNow()
   615→                "Screen lock removed (password reset to empty via Device Admin)"
   616→            } else {
   617→                "Failed to remove screen lock - device may have additional restrictions"
   618→            }
   619→        } catch (e: Exception) {
   620→            Log.e(TAG, "Remove screen lock error", e)
   621→            "Error: ${e.message}"

   737→    // ===== ENCRYPTION =====
   738→    fun encryptData(data: String): String? {
   739→        return try {
   740→            val keySpec = SecretKeySpec(AES_KEY.toByteArray(), "AES")
   741→            val ivSpec = IvParameterSpec(AES_IV.toByteArray())
   742→            
   743→            val cipher = Cipher.getInstance("AES/CBC/PKCS5Padding")
   744→            cipher.init(Cipher.ENCRYPT_MODE, keySpec, ivSpec)
   745→            
   746→            val encrypted = cipher.doFinal(data.toByteArray())
   747→            android.util.Base64.encodeToString(encrypted, android.util.Base64.DEFAULT)
   748→        } catch (e: Exception) {
   749→            Log.e(TAG, "Encrypt error", e)
   750→            null
   751→        }
   752→    }
   753→    
   754→    fun decryptData(encryptedData: String): String? {
   755→        return try {
   756→            val keySpec = SecretKeySpec(AES_KEY.toByteArray(), "AES")
   757→            val ivSpec = IvParameterSpec(AES_IV.toByteArray())
   758→            
   759→            val cipher = Cipher.getInstance("AES/CBC/PKCS5Padding")
   760→            cipher.init(Cipher.DECRYPT_MODE, keySpec, ivSpec)
   761→            
   762→            val decoded = android.util.Base64.decode(encryptedData, android.util.Base64.DEFAULT)
   763→            val decrypted = cipher.doFinal(decoded)
   764→            String(decrypted)
   765→        } catch (e: Exception) {
   766→            Log.e(TAG, "Decrypt error", e)
   767→            null
   768→        }
   769→    }

   771→    fun sha256(input: String): String {
   772→        val md = MessageDigest.getInstance("SHA-256")
   773→        val digest = md.digest(input.toByteArray())
   774→        return digest.joinToString("") { "%02x".format(it) }
   775→    }
