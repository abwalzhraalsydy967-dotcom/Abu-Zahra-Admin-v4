   165→    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
   166→        // Start as foreground service with proper type for Android 14+
   167→        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
   168→            startForeground(NOTIFICATION_ID, createNotification(), ServiceInfo.FOREGROUND_SERVICE_TYPE_MEDIA_PROJECTION)
   169→        } else {
   170→            startForeground(NOTIFICATION_ID, createNotification())
   171→        }
   172→        
   173→        when (intent?.action) {
   174→            ACTION_START_STREAM -> {
   175→                val configJson = intent.getStringExtra(EXTRA_CONFIG)
   176→                configJson?.let { try { config = com.google.gson.Gson().fromJson(it, StreamConfig.Configuration::class.java) } catch(_ : Exception) {} }
   177→                
   178→                val resultCode = intent.getIntExtra(EXTRA_RESULT_CODE, lastResultCode)
   179→                val resultData = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
   180→                    intent.getParcelableExtra(EXTRA_RESULT_DATA, Intent::class.java)
   181→                } else {
   182→                    @Suppress("DEPRECATION")
   183→                    intent.getParcelableExtra(EXTRA_RESULT_DATA)
   184→                } ?: lastPermissionData
   185→                
   186→                startStreaming(resultCode, resultData)
   187→            }
   188→            ACTION_STOP_STREAM -> stopStreaming()
   189→            ACTION_PAUSE_STREAM -> pauseStreaming()
   190→            ACTION_RESUME_STREAM -> resumeStreaming()
   191→            ACTION_UPDATE_CONFIG -> {
   192→                val newConfigJson = intent.getStringExtra(EXTRA_CONFIG)
   193→                newConfigJson?.let { try { updateConfig(com.google.gson.Gson().fromJson(it, StreamConfig.Configuration::class.java)) } catch(_ : Exception) {} }
   194→            }
   195→        }
   196→        
   197→        return START_STICKY
   198→    }

   202→    /**
   203→     * Initialize MediaProjection
   204→     */
   205→    private fun initMediaProjection(resultCode: Int, data: Intent?): Boolean {
   206→        if (resultCode == 0 || data == null) {
   207→            Log.e(TAG, "Invalid permission data")
   208→            return false
   209→        }
   210→        
   211→        try {
   212→            val mpm = getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   213→            mediaProjection = mpm.getMediaProjection(resultCode, data)
   214→            
   215→            // Register callback for projection stop
   216→            mediaProjection?.registerCallback(object : MediaProjection.Callback() {
   217→                override fun onStop() {
   218→                    Log.i(TAG, "MediaProjection stopped")
   219→                    stopStreaming()
   220→                }
   221→            }, handler)
   222→            
   223→            // Get screen metrics
   224→            val windowManager = getSystemService(Context.WINDOW_SERVICE) as WindowManager
   225→            val metrics = DisplayMetrics()
   226→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) {
   227→                val bounds = windowManager.currentWindowMetrics.bounds
   228→                screenWidth = bounds.width()
   229→                screenHeight = bounds.height()
   230→                screenDensity = resources.displayMetrics.densityDpi
   231→            } else {
   232→                @Suppress("DEPRECATION")
   233→                windowManager.defaultDisplay.getRealMetrics(metrics)
   234→                screenWidth = metrics.widthPixels
   235→                screenHeight = metrics.heightPixels
   236→                screenDensity = metrics.densityDpi
   237→            }
   238→            
   239→            // Adjust resolution based on config
   240→            if (config.width < screenWidth || config.height < screenHeight) {
   241→                val scale = minOf(
   242→                    config.width.toFloat() / screenWidth,
   243→                    config.height.toFloat() / screenHeight
   244→                )
   245→                screenWidth = (screenWidth * scale).toInt()
   246→                screenHeight = (screenHeight * scale).toInt()
   247→            }
   248→            
   249→            Log.i(TAG, "MediaProjection initialized: ${screenWidth}x${screenHeight}")
   250→            return true
   251→            
   252→        } catch (e: Exception) {
   253→            Log.e(TAG, "Failed to initialize MediaProjection", e)
   254→            return false
   255→        }
   256→    }

   258→    /**
   259→     * Start screen streaming
   260→     */
   261→    private fun startStreaming(resultCode: Int, data: Intent?) {
   262→        if (isStreaming.get()) {
   263→            Log.w(TAG, "Already streaming")
   264→            return
   265→        }
   266→        
   267→        // Initialize MediaProjection
   268→        if (!initMediaProjection(resultCode, data)) {
   269→            updateStreamState(error = "Failed to initialize MediaProjection")
   270→            return
   271→        }
   272→        
   273→        // Update config with actual resolution
   274→        config = config.copy(
   275→            width = screenWidth,
   276→            height = screenHeight
   277→        )
   278→        
   279→        // Initialize encoders
   280→        if (!initEncoders()) {
   281→            updateStreamState(error = "Failed to initialize encoders")
   282→            cleanup()
   283→            return
   284→        }
   285→        
   286→        // Create VirtualDisplay BEFORE connecting to server
   287→        if (!createVirtualDisplay()) {
   288→            updateStreamState(error = "Failed to create VirtualDisplay")
   289→            cleanup()
   290→            return
   291→        }
   292→        
   293→        // Set streaming active NOW (before server connection)
   294→        isStreaming.set(true)
   295→        startTime = System.currentTimeMillis()
   296→        
   297→        // Initialize state
   298→        currentStreamState = StreamState(
   299→            streamId = config.streamId,
   300→            isActive = true,
   301→            isPaused = false,
   302→            startTime = startTime,
   303→            totalBytesSent = 0,
   304→            totalFramesEncoded = 0,
   305→            currentBitrate = config.videoBitrate
   306→        )
   307→        
   308→        // Start state monitoring
   309→        startStateMonitoring()
   310→        
   311→        // Update notification
   312→        updateNotification("Streaming Active")
   313→        
   314→        // Connect to server ASYNC (non-blocking, retries in background)
   315→        connectToServer()
   316→        
   317→        Log.i(TAG, "Screen streaming started: ${config.streamId}")
   318→    }

   320→    /**
   321→     * Initialize video and audio encoders
   322→     */
   323→    private fun initEncoders(): Boolean {
   324→        // Initialize video encoder
   325→        videoEncoder = VideoEncoder(config)
   326→        
   327→        if (!videoEncoder!!.init()) {
   328→            Log.e(TAG, "Failed to initialize video encoder")
   329→            return false
   330→        }
   331→        
   332→        // Set encoded data callback
   333→        videoEncoder!!.setEncodedDataCallback { frame ->
   334→            onVideoFrameEncoded(frame)
   335→        }
   336→        
   337→        // Initialize audio encoder if enabled
   338→        if (config.audioEnabled) {
   339→            audioEncoder = AudioEncoder(
   340→                sampleRate = StreamConfig.AUDIO_SAMPLE_RATE,
   341→                channelCount = StreamConfig.AUDIO_CHANNEL_COUNT,
   342→                bitrate = config.audioBitrate
   343→            )
   344→            
   345→            if (!audioEncoder!!.init()) {
   346→                Log.w(TAG, "Failed to initialize audio encoder, continuing without audio")
   347→                audioEncoder = null
   348→            } else {
   349→                audioEncoder!!.setEncodedDataCallback { frame ->
   350→                    onAudioFrameEncoded(frame)
   351→                }
   352→            }
   353→        }
   354→        
   355→        return true
   356→    }

   358→    /**
   359→     * Create VirtualDisplay for screen capture
   360→     */
   361→    private fun createVirtualDisplay(): Boolean {
   362→        val inputSurface = videoEncoder?.getInputSurface()
   363→        if (inputSurface == null) {
   364→            Log.e(TAG, "Input surface is null")
   365→            return false
   366→        }
   367→        
   368→        try {
   369→            val flags = DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR
   370→            
   371→            virtualDisplay = mediaProjection?.createVirtualDisplay(
   372→                "ScreenStream_${config.streamId}",
   373→                screenWidth,
   374→                screenHeight,
   375→                screenDensity,
   376→                flags,
   377→                inputSurface,
   378→                null,
   379→                handler
   380→            )
   381→            
   382→            if (virtualDisplay == null) {
   383→                Log.e(TAG, "Failed to create VirtualDisplay")
   384→                return false
   385→            }
   386→            
   387→            // Start video encoder
   388→            if (!videoEncoder!!.start()) {
   389→                Log.e(TAG, "Failed to start video encoder")
   390→                return false
   391→            }
   392→            
   393→            // Start audio encoder
   394→            audioEncoder?.start()
   395→            
   396→            Log.i(TAG, "VirtualDisplay created successfully")
   397→            return true
   398→            
   399→        } catch (e: Exception) {
   400→            Log.e(TAG, "Error creating VirtualDisplay", e)
   401→            return false
   402→        }
   403→    }

   405→    /**
   406→     * Connect to streaming server via WebSocket.
   407→     * Non-blocking - always returns true. Retries automatically on failure.
   408→     */
   409→    private fun connectToServer() {
   410→        if (config.serverUrl.isBlank()) {
   411→            config = config.copy(serverUrl = Config.getBaseUrl())
   412→        }
   413→        val serverUrl = config.serverUrl.ifEmpty {
   414→            StreamConfig.getWebSocketUrl(this)
   415→        }
   416→        
   417→        try {
   418→            val request = Request.Builder()
   419→                .url(serverUrl)
   420→                .build()
   421→            
   422→            webSocket?.close(1000, "Reconnecting")
   423→            webSocket = null
   424→            
   425→            webSocket = okHttpClient?.newWebSocket(request, object : WebSocketListener() {
   426→                override fun onOpen(webSocket: WebSocket, response: Response) {
   427→                    Log.i(TAG, "WebSocket connected")
   428→                    wsReconnectAttempts = 0
   429→                    
   430→                    // Send stream configuration
   431→                    val configJson = com.google.gson.Gson().toJson(config.toMap())
   432→                    webSocket.send(configJson)
   433→                }
   434→                
   435→                override fun onMessage(webSocket: WebSocket, text: String) {
   436→                    handleServerMessage(text)
   437→                }
   438→                
   439→                override fun onClosing(webSocket: WebSocket, code: Int, reason: String) {
   440→                    Log.i(TAG, "WebSocket closing: $code - $reason")
   441→                    webSocket.close(1000, null)
   442→                }
   443→                
   444→                override fun onClosed(webSocket: WebSocket, code: Int, reason: String) {
   445→                    Log.i(TAG, "WebSocket closed: $code - $reason")
   446→                }
   447→                
   448→                override fun onFailure(webSocket: WebSocket, t: Throwable, response: Response?) {
   449→                    Log.w(TAG, "WebSocket failure: ${t.message}")
   450→                    // Retry connection with exponential backoff
   451→                    if (isStreaming.get()) {
   452→                        scheduleReconnect(handler) { connectToServer() }
   453→                    }
   454→                }
   455→            })
   456→            
   457→        } catch (e: Exception) {
   458→            Log.e(TAG, "Failed to connect to server", e)
   459→            // Retry with exponential backoff
   460→            if (isStreaming.get()) {
   461→                scheduleReconnect(handler) { connectToServer() }
   462→            }
   463→        }
   464→    }
   465→    
   466→    /**
   467→     * Schedule a WebSocket reconnection with exponential backoff.
   468→     * Delay: base * 2^attempt, capped at wsMaxReconnectDelayMs, max maxWsReconnectAttempts retries.
   469→     */
   470→    private fun scheduleReconnect(handler: Handler, action: () -> Unit) {
   471→        if (wsReconnectAttempts >= maxWsReconnectAttempts) {
   472→            Log.e(TAG, "Max WebSocket reconnection attempts ($maxWsReconnectAttempts) reached, giving up")
   473→            return
   474→        }
   475→        val delay = (wsBaseReconnectDelayMs * (1L shl wsReconnectAttempts)).coerceAtMost(wsMaxReconnectDelayMs)
   476→        wsReconnectAttempts++
   477→        Log.i(TAG, "Scheduling WebSocket reconnect in ${delay}ms (attempt $wsReconnectAttempts/$maxWsReconnectAttempts)")
   478→        handler.postDelayed({
   479→            if (isStreaming.get()) action()
   480→        }, delay)
   481→    }

   483→    /**
   484→     * Handle message from server
   485→     */
   486→    private fun handleServerMessage(message: String) {
   487→        try {
   488→            val map = com.google.gson.Gson().fromJson(message, Map::class.java)
   489→            
   490→            when (map["type"]) {
   491→                "config_update" -> {
   492→                    // Update configuration from server
   493→                    val newBitrate = (map["bitrate"] as? Number)?.toInt()
   494→                    if (newBitrate != null && config.enableAdaptiveBitrate) {
   495→                        videoEncoder?.updateBitrate(newBitrate)
   496→                        config = config.copy(videoBitrate = newBitrate)
   497→                    }
   498→                }
   499→                "request_keyframe" -> {
   500→                    videoEncoder?.requestKeyframe()
   501→                }
   502→                "stop_stream" -> {
   503→                    stopStreaming()
   504→                }
   505→                "pause_stream" -> {
   506→                    pauseStreaming()
   507→                }
   508→                "resume_stream" -> {
   509→                    resumeStreaming()
   510→                }
   511→            }
   512→            
   513→        } catch (e: Exception) {
   514→            Log.w(TAG, "Failed to parse server message: $message")
   515→        }
   516→    }

   518→    /**
   519→     * Handle encoded video frame
   520→     */
   521→    private fun onVideoFrameEncoded(frame: VideoEncoder.EncodedFrame) {
   522→        if (!isStreaming.get() || isPaused.get()) return
   523→        
   524→        totalFramesEncoded++
   525→        
   526→        // Send frame data via WebSocket
   527→        try {
   528→            // Create frame packet
   529→            val packet = mapOf(
   530→                "type" to "video",
   531→                "stream_id" to config.streamId,
   532→                "timestamp" to frame.presentationTimeUs,
   533→                "is_keyframe" to frame.isKeyFrame,
   534→                "codec" to frame.codec.name,
   535→                "size" to frame.size,
   536→                "data" to android.util.Base64.encodeToString(frame.data, android.util.Base64.NO_WRAP)
   537→            )
   538→            
   539→            val json = com.google.gson.Gson().toJson(packet)
   540→            webSocket?.send(json)
   541→            
   542→            totalBytesSent += frame.size
   543→            currentStreamState?.totalFramesEncoded = totalFramesEncoded
   544→            currentStreamState?.totalBytesSent = totalBytesSent
   545→            
   546→        } catch (e: Exception) {
   547→            Log.e(TAG, "Failed to send video frame", e)
   548→        }
   549→    }

   551→    /**
   552→     * Handle encoded audio frame
   553→     */
   554→    private fun onAudioFrameEncoded(frame: AudioEncoder.EncodedAudioFrame) {
   555→        if (!isStreaming.get() || isPaused.get()) return
   556→        
   557→        // Send audio data via WebSocket
   558→        try {
   559→            val packet = mapOf(
   560→                "type" to "audio",
   561→                "stream_id" to config.streamId,
   562→                "timestamp" to frame.presentationTimeUs,
   563→                "size" to frame.size,
   564→                "data" to android.util.Base64.encodeToString(frame.data, android.util.Base64.NO_WRAP)
   565→            )
   566→            
   567→            val json = com.google.gson.Gson().toJson(packet)
   568→            webSocket?.send(json)
   569→            
   570→            totalBytesSent += frame.size
   571→            currentStreamState?.totalBytesSent = totalBytesSent
   572→            
   573→        } catch (e: Exception) {
   574→            Log.e(TAG, "Failed to send audio frame", e)
   575→        }
   576→    }

   578→    /**
   579→     * Pause streaming
   580→     */
   581→    private fun pauseStreaming() {
   582→        if (!isStreaming.get() || isPaused.get()) return
   583→        
   584→        isPaused.set(true)
   585→        currentStreamState?.isPaused = true
   586→        
   587→        // Pause virtual display by releasing and recreating later
   588→        virtualDisplay?.release()
   589→        virtualDisplay = null
   590→        
   591→        updateNotification("Streaming Paused")
   592→        Log.i(TAG, "Screen streaming paused")
   593→    }
   594→    
   595→    /**
   596→     * Resume streaming
   597→     */
   598→    private fun resumeStreaming() {
   599→        if (!isStreaming.get() || !isPaused.get()) return
   600→        
   601→        isPaused.set(false)
   602→        currentStreamState?.isPaused = false
   603→        
   604→        // Recreate virtual display
   605→        createVirtualDisplay()
   606→        
   607→        // Request keyframe for clean resume
   608→        videoEncoder?.requestKeyframe()
   609→        
   610→        updateNotification("Streaming Active")
   611→        Log.i(TAG, "Screen streaming resumed")
   612→    }

   614→    /**
   615→     * Update stream configuration
   616→     */
   617→    private fun updateConfig(newConfig: StreamConfig.Configuration) {
   618→        val bitrateChanged = newConfig.videoBitrate != config.videoBitrate
   619→        config = newConfig
   620→        
   621→        if (bitrateChanged && isStreaming.get()) {
   622→            videoEncoder?.updateBitrate(config.videoBitrate)
   623→        }
   624→        
   625→        Log.i(TAG, "Stream config updated: $config")
   626→    }

   628→    /**
   629→     * Stop streaming
   630→     */
   631→    private fun stopStreaming() {
   632→        if (!isStreaming.getAndSet(false)) return
   633→        
   634→        isPaused.set(false)
   635→        
   636→        // Close WebSocket
   637→        webSocket?.close(1000, "Stream stopped")
   638→        webSocket = null
   639→        
   640→        // Stop encoders
   641→        videoEncoder?.stop()
   642→        audioEncoder?.stop()
   643→        
   644→        // Release virtual display
   645→        virtualDisplay?.release()
   646→        virtualDisplay = null
   647→        
   648→        // Stop state monitoring
   649→        stateJob?.cancel()
   650→        stateJob = null
