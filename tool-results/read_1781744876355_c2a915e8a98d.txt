   630→// ── LOGIN / LOGOUT ──
   631→function doLogin() {
   632→  var server = document.getElementById('loginServer').value.replace(/\/+$/, '');
   633→  var user = document.getElementById('loginUser').value.trim();
   634→  var pass = document.getElementById('loginPass').value;
   635→  if (!server || !user || !pass) {
   636→    document.getElementById('loginError').textContent = 'يرجى ملء جميع الحقول';

   638→  }
   639→  var btn = document.getElementById('loginBtn');

   641→  btn.textContent = 'جاري تسجيل الدخول...';
   642→  fetch(server + '/api/web/login', {
   643→    method: 'POST',
   644→    headers: {'Content-Type': 'application/json'},
   645→    body: JSON.stringify({username: user, password: pass})
   646→  })
   647→  .then(function(r) { return r.json(); })
   648→  .then(function(data) {
   649→    btn.disabled = false;
   650→    btn.textContent = 'تسجيل الدخول';
   651→    if (data.ok && data.token) {
   652→      APP.token = data.token;
   653→      APP.server = server;
   654→      APP.username = data.username;
   655→      APP.role = data.role;
   656→      localStorage.setItem('az_token', data.token);

   658→      localStorage.setItem('az_username', data.username);
   659→      localStorage.setItem('az_role', data.role);

   661→    } else {
   662→      document.getElementById('loginError').textContent = data.message || 'فشل تسجيل الدخول';
   663→    }
   664→  })
   665→  .catch(function(err) {
   666→    btn.disabled = false;
   667→    btn.textContent = 'تسجيل الدخول';

   669→  });
   670→}

   746→// ── WEBSOCKET ──
   747→function connectWS() {
   748→  if (APP.ws) { try { APP.ws.close(); } catch(e){} }
   749→  var protocol = APP.server.startsWith('https') ? 'wss' : 'ws';
   750→  var wsUrl = protocol + '://' + APP.server.replace(/^https?:\/\//, '') + '/ws/dashboard?token=' + APP.token;
   751→  try {
   752→    APP.ws = new WebSocket(wsUrl);
   753→    APP.ws.onopen = function() {
   754→      APP.wsRetries = 0;
   755→      document.getElementById('wsDot').classList.add('connected');
   756→      document.getElementById('wsText').textContent = 'متصل';
   757→    };
   758→    APP.ws.onmessage = function(evt) {
   759→      try {
   760→        var msg = JSON.parse(evt.data);
   761→        handleWSMessage(msg);
   762→      } catch(e) {}
   763→    };
   764→    APP.ws.onclose = function() {
   765→      document.getElementById('wsDot').classList.remove('connected');
   766→      document.getElementById('wsText').textContent = 'غير متصل';
   767→      reconnectWS();
   768→    };
   769→    APP.ws.onerror = function() {
   770→      APP.ws.close();
   771→    };
   772→  } catch(e) {
   773→    reconnectWS();
   774→  }
   775→}
   776→
   777→function reconnectWS() {
   778→  if (APP.wsRetries >= APP.wsMaxRetries) return;
   779→  APP.wsRetries++;
   780→  var delay = Math.min(1000 * Math.pow(2, APP.wsRetries), 30000);
   781→  setTimeout(function() { connectWS(); }, delay);
   782→}

   784→function handleWSMessage(msg) {
   785→  if (msg.type === 'init') {
   786→    if (msg.devices) { APP.devices = msg.devices; populateDeviceSelectors(); }
   787→    if (msg.stats) { APP.stats = msg.stats; renderStats(); }
   788→    if (msg.commands) { APP.commands = msg.commands; }
   789→    if (msg.categories) { APP.categories = msg.categories; renderCmdTabs(); }
   790→    renderDashActiveDevices();
   791→  } else if (msg.type === 'stats_update') {

   893→function formatUptime(secs) {
   894→  if (!secs || secs < 0) return '0 د';
   895→  var d = Math.floor(secs / 86400);
   896→  var h = Math.floor((secs % 86400) / 3600);
   897→  var m = Math.floor((secs % 3600) / 60);
   898→  if (d > 0) return d + 'ي ' + h + 'س';
   899→  if (h > 0) return h + 'س ' + m + 'د';
   900→  return m + 'د';
   901→}

  1459→function formatSize(bytes) {
  1460→  if (!bytes || bytes === 0) return '0 B';
  1461→  var sizes = ['B', 'KB', 'MB', 'GB'];
  1462→  var i = Math.floor(Math.log(bytes) / Math.log(1024));
  1463→  i = Math.min(i, sizes.length - 1);
  1464→  return (bytes / Math.pow(1024, i)).toFixed(1) + ' ' + sizes[i];
  1465→}
