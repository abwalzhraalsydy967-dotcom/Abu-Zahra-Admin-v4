    32→@web.middleware
    33→async def cors_middleware(request, handler):
    34→    allowed_origins = ['https://alsydyabwalzhra.online', 'http://localhost:8443']
    35→    origin = request.headers.get('Origin', '')
    36→    if request.method == 'OPTIONS':
    37→        response = web.Response(status=204)
    38→        if origin in allowed_origins:
    39→            response.headers['Access-Control-Allow-Origin'] = origin
    40→            response.headers['Access-Control-Allow-Methods'] = 'GET, POST, PUT, DELETE, OPTIONS'
    41→            response.headers['Access-Control-Allow-Headers'] = 'Authorization, Content-Type, X-Device-Token'
    42→            response.headers['Access-Control-Allow-Credentials'] = 'true'
    43→        return response
    44→    response = await handler(request)
    45→    if origin in allowed_origins:
    46→        response.headers['Access-Control-Allow-Origin'] = origin

    48→        response.headers['Access-Control-Allow-Headers'] = 'Authorization, Content-Type, X-Device-Token'
    49→        response.headers['Access-Control-Allow-Credentials'] = 'true'
    50→    return response

   342→def _get_real_ip(request):
   343→    """Extract real client IP from request, handling reverse proxy headers."""
   344→    forwarded = request.headers.get("X-Forwarded-For", "").strip()
   345→    if forwarded:
   346→        return forwarded.split(",")[0].strip()
   347→    real_ip = request.headers.get("X-Real-IP", "").strip()
   348→    if real_ip:
   349→        return real_ip
   350→    return request.remote or ""

   439→def format_uptime(seconds):
   440→    d = seconds // 86400
   441→    h = (seconds % 86400) // 3600
   442→    m = (seconds % 3600) // 60
   443→    s = seconds % 60
   444→    parts = []
   445→    if d: parts.append(f"{d}d")
   446→    if h: parts.append(f"{h}h")
   447→    if m: parts.append(f"{m}m")
   448→    parts.append(f"{s}s")
   449→    return " ".join(parts)

   530→    async with _db_lock:
   531→        commands = load_json(COMMANDS_FILE, [])

   537→            "status": "pending",
   538→            "created_at": ts(),
   539→            "sent_at": None,
   540→            "result": None,
   541→        }
   542→        commands.append(cmd)
   543→        if len(commands) > 1000:
   544→            commands = commands[-1000:]
   545→        save_json(COMMANDS_FILE, commands)
   546→        append_event("Command queued", {"device_id": device_id, "command": command, "cmd_id": cmd["id"]})
   547→        firebase_push_command(cmd)
   548→        return cmd

   572→async def create_session(username, password, ip="", ua=""):
   573→    async with _db_lock:
   574→        settings = load_settings()
   575→        if not hmac.compare_digest(password, settings.get("admin_password", "admin")):
   576→            return None
   577→        # Clean expired sessions first
   578→        sessions = load_json(SESSIONS_FILE, [])
   579→        now = datetime.now(timezone.utc)
   580→        active = []
   581→        for s in sessions:
   582→            try:
   583→                if now < datetime.fromisoformat(s.get("expires_at", "")).replace(tzinfo=timezone.utc):

   675→async def firebase_set(path, data):
   676→    """SET data in Firebase RTDB - with optional Database Secret auth.
   677→    If data is None, uses DELETE instead of PUT (to properly remove the key).
   678→    Gracefully returns False if Firebase is unavailable."""
   679→    try:
   680→        url = f"{FIREBASE_RTDB_URL}/{path}.json"
   681→        if FIREBASE_DB_SECRET:
   682→            url += f"?auth={FIREBASE_DB_SECRET}"
   683→        session = get_tg_session()
   684→        if data is None:
   685→            # حذف المسار بدلاً من تعيينه إلى null
   686→            async with session.delete(url, timeout=aiohttp.ClientTimeout(total=10)) as resp:
   687→                ok = resp.status in (200, 204)
   688→                if not ok:
   689→                    body = await resp.text()
   690→                    log.warning("Firebase DELETE %s failed: status=%d body=%s", path, resp.status, body[:200])
   691→                return ok
   692→        else:
   693→            async with session.put(url, json=data, timeout=aiohttp.ClientTimeout(total=10)) as resp:
   694→                ok = resp.status in (200, 204)
   695→                if ok:
   696→                    log.debug("Firebase SET %s OK", path)
   697→                else:
   698→                    body = await resp.text()
   699→                    log.warning("Firebase SET %s failed: status=%d body=%s", path, resp.status, body[:200])
   700→                return ok
   701→    except Exception as exc:
   702→        log.error("Firebase SET %s failed: %s", path, exc)
   703→        return False

   743→async def _firebase_push_cmd_async(cmd):
   744→    """Async: Push command to Firebase /commands/{device_id}/{cmd_id}"""
   745→    device_id = cmd.get("device_id", "")
   746→    cmd_id = cmd.get("id", "")
   747→    if not device_id or not cmd_id:

   753→            "id": cmd["id"],
   754→            "device_id": cmd["device_id"],

   758→            "created_at": cmd["created_at"],

   760→            "server_port": SERVER_PORT,
   761→        })
   762→        if ok:
   763→            log.info("Firebase: Command %s pushed for device %s", cmd_id, device_id)

   770→                except Exception:
   771→                    pass
   772→            asyncio.ensure_future(_delayed_delete())
   773→        else:
   774→            log.warning("Firebase: Failed to push command %s", cmd_id)

   776→        log.error("Firebase push command error: %s", exc)

   870→async def send_message(chat_id, text, parse_mode="HTML", reply_markup=None, disable_notification=False):
   871→    global messages_sent
   872→    # === ANTI-SPAM: Deduplication + Rate Limiting ===
   873→    now = time.time()
   874→    chat_key = str(chat_id)
   875→    text_hash = hashlib.md5(text[:200].encode()).hexdigest() if text else ""
   876→
   877→    # Check: same message to same chat within 60 seconds?
   878→    dedup_key = f"{chat_key}:{text_hash}"
   879→    last_sent = _message_dedup.get(dedup_key, 0)
   880→    if now - last_sent < 10 and text_hash:
   881→        log.warning("DEDUP BLOCKED: chat=%s hash=%s", chat_key, text_hash[:8])
   882→        return {"ok": False, "description": "dedup_blocked"}
   883→
   884→    # Check: more than 5 messages to same chat within 30 seconds?
   885→    recent = _chat_rate_counter.get(chat_key, [])
   886→    recent = [t for t in recent if now - t < 30]
   887→    if len(recent) >= 10:
   888→        log.warning("RATE BLOCKED: chat=%s count=%d", chat_key, len(recent))
   889→        return {"ok": False, "description": "rate_limited"}
   890→    recent.append(now)
   891→    _chat_rate_counter[chat_key] = recent
   892→
   893→    # Register in dedup table
   894→    _message_dedup[dedup_key] = now
   895→    expired = [k for k, v in _message_dedup.items() if now - v > 30]
   896→    for k in expired:
   897→        del _message_dedup[k]
   898→
   899→    payload = {"chat_id": chat_id, "text": text, "parse_mode": parse_mode}
   900→    if reply_markup:
   901→        payload["reply_markup"] = reply_markup
   902→    if disable_notification:
   903→        payload["disable_notification"] = True
   904→    result = await tg_request("sendMessage", payload)
   905→    if result and result.get("ok"):
   906→        messages_sent += 1
   907→    return result

   997→def check_user_rate_limit(chat_id):

  1004→    if len(timestamps) >= USER_RATE_LIMIT_MAX:
  1005→        _user_command_timestamps[chat_key] = timestamps

  1009→    # Periodic cleanup
  1010→    if len(timestamps) == 1:
  1011→        expired_keys = [k for k, v in _user_command_timestamps.items() if not v or now - v[-1] > USER_RATE_LIMIT_WINDOW * 2]
  1012→        for k in expired_keys:
  1013→            del _user_command_timestamps[k]
  1014→    return True
